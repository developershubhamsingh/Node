# start mongodb server : 
> open cmd and start server (27017)
> cd C:\Program Files\MongoDB\Server\7.0\bin
> mongod
> Do not close this cmd

# start mongo client : 
> open cmd again and start server (27017)
> cd C:\Program Files\MongoDB\Server\7.0\bin
> mongosh
